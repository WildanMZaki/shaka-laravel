berikut catatan untuk konfigurasi umum datatable dengan serverside
table = $('#data-table-simple').DataTable({ 
	"language": {                
        "infoFiltered": "", 
        processing: '<div class="loader-product">Loading&#8230;</div>'
   	},
	"processing": true, 
    "serverSide": true, 
    "bDestroy": true,
    "order": [], 
     
    "ajax": {
        "url": 'report/mutasi/get_data',
        "type": "POST", 
        "data": function (data){
            data.start_date = $('#start_date').val();
            data.end_date = $('#end_date').val();
            data.code_toko = $('#code_toko').val();
      	}, 
        "error": function (jqXHR, textStatus, errorThrown) {
                Materialize.toast("Data Tidak Ditemukan !", 2500); 
                setTimeout(function() {
					location.reload();
				}, 1000);
            }
        },
          
        "columnDefs": [
    		{ 
        		"targets": [ 0 ], 
        		"orderable": false, 
        	},
      	],
});
